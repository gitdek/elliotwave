  8 #' linked_list
  9 #'
 10 #' @return function list
 11 #'
 12 linked_list <- function() {
 13   head <- list(0)
 14   length <- 0
 15 
 16   methods <- list()
 17 
 18   methods$add <- function(val) {
 19     length <<- length + 1
 20     head <<- list(head, val)
 21   }
 22 
 23   methods$as.list <- function() {
 24     b <- vector('list', length)
 25     h <- head
 26     for (i in length:1) {
 27       b[[i]] <- head[[2]]
 28       head <- head[[1]]
 29     }
 30     return(b)
 31   }
 32   methods
 33 }

 35 #' @title Golden Ratio
 36 #' @description Determine a local minimum of a continuous real function f
 37 #'
 38 #' @param f function
 39 #' @param a numeric
 40 #' @param b numeric > a
 41 #' @param eps numeric (default 1e-16)
 42 #' @param maxiter  maximum number of iterations
 43 #' @return  list containing approximation of local minimum,
 44 #'   value at local minimum and some control data
 45 #' @export
 46 golden_ratio <- function(f,
 47                          a,
 48                          b,
 49                          eps = 1e-16,
 50                          maxiter = 100) {
 51   stopifnot(
 52     is.numeric(a) & is.finite(a),
 53     is.numeric(b) & is.finite(b),
 54     b > a,
 55     is.numeric(eps) & is.finite(eps) & eps > 0,
 56     is.numeric(maxiter) & is.finite(maxiter) & maxiter > 0
 57   )
 58 
 59   phi <- (sqrt(5) - 1) / 2
 60   convergence <- 1
 61 
 62   for (i in 1:maxiter) {
 63     # Update xl, xp
 64     xl <- b - phi * (b - a)
 65     xp <- a + phi * (b - a)
 66     if (f(xl) > f(xp)) {
 67       a <- xl
 68     } else {
 69       b <- xp
 70     }
 71 
 72     # Check if method converged
 73     if (abs(b - a) < eps) {
 74       convergence <- 0
 75       break
 76     }
 77   }
 78 
 79   if (convergence == 1)
 80     warning("Method didn't converge")
 81 
 82   list(
 83     par = (a + b) / 2,
 84     value = f((a + b) / 2),
 85     counts = i,
 86     convergence = convergence,
 87     message = NULL
 88   )
 89 }

 92 #' @title  Elliot wave - puglisij
 93 #' @description  Criteria must match as follows: Climb in instrument begins as wave 1, followed by a retracement
 94 #' which is wave 2, followed by another climb wave 3, followed by a retracement wave 4, and
 95 #' then followed by the final climb, wave 5.
 96 #'
 97 #' @param v  vector
 98 #' @param eps  numeric
 99 #' @param maxiter  maximum number of iterations
100 #' @param fr  fib ratio
101 #' @return list containing entries and golden ratio computed position scaled
102 #' @author puglisij
103 #' @details Wave 1 starts at a valley
104 #' Wave 2 can never drop lower in price than the start of Wave 1
105 #' 
106 #' Wave 2 lies in time interval , where =0.382*
107 #' 
108 #' Wave 2 bounces of a Fibonacci retracement level
109 #' 
110 #' Wave 3 can never be the shortest impulse wave
111 #' 
112 #' Wave 3 lies in time interval , where =1.618*
113 #' 
114 #' Wave 4 can never drop lower in price than the end of Wave 1
115 #' 
116 #' Wave 4 lies in time interval , where =0.382*
117 #' 
118 #' Wave 5 lies in time interval , where =1.618*
119 #' 
120 #' Wave 5 is not obliged to peak beyond the end of Wave 3 in price, this is called truncation
121 #' 
122 #' Inverted criteria apply to inverted Elliott wave patterns
123 #' @export
124 #'
125 elliot_wave <- function(v,
126                         eps = 1e-16,
127                         maxiter = 10000,
128                         fr = 0.236) {
129   stopifnot(is.vector(v))
130   x <- vector(numeric, length = length(v))
131   j <- as.vector(v)
132   z <- vector(mode = "complex", length = 0)
133   peaks_valleys <- cbind(x, z)
134   entrys <- linked_list()
135 
136   for (i in 2:length(v) - 1) {
137     if (v(i) <= v(i + 1) &&
138         v(i - 1) >= v(i) || v(i) >= v(i + 1) && v(i - 1) <= v(i)) {
139       x = c(x, v(i))
140       z = c(z, j(i))
141 
142       x = x[1, diff(x) != 0]
143       z = z[1, diff(x) != 0]
144       peaks_valleys = c(x, z)
145     }
146   }
147 
148   for (n in 3:length(x) - 1) {
149     if (x(n) < x(n - 1) &&
150         x(n) >= x(n - 2) &&
151         x(n + 1) > x(n - 1) && x(n + 2) > x(n - 1)) {
152       if (x(n + 1) - x(n) >= x(n - 1) - x(n - 2) || x(n + 1) - x(n) >= x
153           (n + 3) - x(n + 2)) {
154         if (z(n) - z(n - 1) <= (0.382 * (z(n - 1) - z(n - 2))) &&
155             z(n + 1) - z(n) <= (1.618 * (z(n - 1) - z(n - 2))) &&
156             z(n + 2) - z(n + 1) <= (0.382 * (z(n + 1) - z(n))) &&
157             z(n + 3) - z(n + 2) <= (1.618 * (z(n + 1) - z(n)))) {
158           if (x(n) <= (x(n - 1) - ((x(n - 1) - x(n - 2)) * fr)) * 1.001 &&
159               x(n) >= (x(n - 1) - ((x(n - 1) - x(n - 2)) * fr)) * 0.999) {
160             entry <-
161               function(fib_ratio,
162                        instrument_vector,
163                        index)
164                 ((0.999 * instrument_vector(index + 3)) - ((
165                   instrument_vector(index - 1) - ((
166                     instrument_vector(index - 1) - instrument_vector(index - 2)
167                   ) * fib_ratio
168                   )
169                 ) * 1.001)) / ((instrument_vector(index - 1) - ((
170                   instrument_vector(index -
171                                       1) - instrument_vector(index - 2)
172                 ) * fib_ratio
173                 )) * 1.001)
174 
175             e <- entry(
176               fib_ratio = fr,
177               instrument_vector = x,
178               index = n
179             )
180 
181             e2 <- entry(
182               fib_ratio = 0.618,
183               instrument_vector = x,
184               index = n
185             )
186 
187             gr <-
188               golden_ratio(
189                 function(x)
190                   x * cos(0.1 * exp(x)) * sin(0.1 * exp(x)),
191                 a = e,
192                 b = e2,
193                 maxiter = maxiter
194               )
195 
196             entrys$add(e, e2, gr)
197 
198           }
199         }
200 
201       }
202     }
203   }
204 
205   entrys
206 }
